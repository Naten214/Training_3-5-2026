az acr create `
--resource-group 20251229-PWC-Azure `
--name --- `
--location --`
--sku Basic 

az container create `
--resource-group 20251229-PWC-Azure `
--name natedemo5container `

--registry-username $ACR_USERNAME `
--registry-password $ACR_PASSWORD `
--restart-policy Never `

--name natecontainerapp `
--resource-group 20251229-PWC-Azure `
--location eastus

--registry-username $ACR_USERNAME `
--registry-password $ACR_PASSWORD `
--cpu 0.5 `
